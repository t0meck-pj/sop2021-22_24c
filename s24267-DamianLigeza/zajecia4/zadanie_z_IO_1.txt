	#include <iostream>
	#include <fstream>
	#include <string>
	#include <vector>

	int main() {
		std::ifstream numbers;
		
		numbers.open("input.txt", std::ios::in);
		
		if (!numbers.is_open()) {
			perror("Can't open file");
			return 1;
		}
		else {
			std::string firstLine;
			
			std::getline(numbers, firstLine);
			int numberOfLines = std::stoi(firstLine);
			
			std::vector<int> numbersVector;
			
			for (int i = 0; i < numberOfLines; i++) {
				std::string line;
				std::getline(numbers, line);
				if (line.size() > 0) {
					numbersVector.push_back(stoi(line));
				}
			}

			for (int i : numbersVector) {
				std::cout << i << std::endl;
			}
		}

		numbers.close();
		return 0;
	}
