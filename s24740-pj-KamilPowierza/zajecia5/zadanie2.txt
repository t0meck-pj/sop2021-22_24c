#include <stdio.h>
#include <unistd.h>
#include <sys/types.h>
#include <sys/wait.h>

int main(){

	pid_t masterPid = getpid();
	pid_t p = fork();
		
	if (p > 0){
		printf("PID: %d , PPID: %d \n", getpid(), getppid());	
	}else{
		printf("PID: %d , PPID: %d \n", getpid(), getppid());
	}

	if(masterPid == getpid()){
		int status = 0;
		waitpid(0, &status, 0);
	}
	
	return 0;
}
